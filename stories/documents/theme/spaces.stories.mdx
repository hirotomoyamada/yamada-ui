## Default

<Provider>
  <Grid templateColumns='5rem 5rem 5rem 1fr' my='md' gap='md'>
    <Text>Name</Text>
    <Text>Rem</Text>
    <Text>Pixel</Text>
    <Box></Box>
    <Text>px</Text>
    <Text>0.0625rem</Text>
    <Text>1px</Text>
    <Box maxW='1px' h='9xs' bg='green.500'></Box>
    {Object.entries(defaultTheme.spaces)
      .filter(([key]) => !isNaN(Number(key)))
      .sort(([a], [b]) => a - b)
      .map(([key, value]) => {
        return (
          <Fragment key={key}>
            <Text>{key}</Text>
            <Text>{value}</Text>
            <Text>{`${parseFloat(value) * 16}px`}</Text>
            <Box maxW={value} h='9xs' bg='green.500'></Box>
          </Fragment>
        )
      })}
  </Grid>
</Provider>
